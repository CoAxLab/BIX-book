Run#

detection_radii = [1, 2, 3, 4, 5]
colors = ["darkgreen", "seagreen", "cadetblue", "steelblue", "mediumpurple"]
names = list(range(5))

# !
results = []
for name, detection_radius in zip(names, detection_radii):
    # Intial targets
    env.add_targets(
        target_index, 
        targets, 
        values, 
        detection_radius=detection_radius, 
        p_target=p_target
    )

    # Preds
    pred = [DiffusionGrid(
            scale=scale, 
            min_length=min_length, 
            step_size=step_size
    )]

    # Preys
    preys = [
        DiffusionGrid(
            scale=scale, 
            min_length=min_length, 
            step_size=step_size
        )] * num_targets

    # All agents
    agents = preys + pred

    # !
    comp_exp = multi_experiment(
        f"test_{name}",
        agents, 
        env,
        num_steps=num_steps,
        num_experiments=num_experiments,
        seed=404,
        split_state=False,
        dump=False
    )
    results.append(comp_exp)
